COOKIES_FILE = os.path.abspath(os.path.join('Cookies', 'facebook_cookies.json'))

def save_cookies(driver):
    os.makedirs('test', exist_ok=True)
    cookies = driver.get_cookies()
    with open(COOKIES_FILE, 'w') as f:
        json.dump(cookies, f, indent=2)
    print('✅ Cookies guardadas en carpeta test')

def load_cookies(driver):
    if not os.path.exists(COOKIES_FILE):
        print('No hay cookies guardadas')
        return False
    
    # Primero ir a Facebook para establecer el dominio
    driver.get('https://www.facebook.com/')
    time.sleep(3)
    
    with open(COOKIES_FILE, 'r') as f:
        cookies = json.load(f)
    
    for cookie in cookies:
        cookie.pop('sameSite', None)
        cookie.pop('expiry', None)  # ← esto causa problemas a veces
        try:
            driver.add_cookie(cookie)
        except Exception as e:
            pass
    
    # Refrescar para aplicar cookies
    driver.refresh()
    time.sleep(5)
    
    print(f'✅ {len(cookies)} cookies cargadas')
    return True

def is_logged_in(driver):
    return 'login' not in driver.current_url and 'checkpoint' not in driver.current_url

def parse_members(text):
    """Convierte '113,6 mil' → 113600"""
    text = text.lower().strip()
    match = re.search(r'([\d,.]+)\s*(mil)?', text)
    if not match:
        return None
    num = float(match.group(1).replace(',', '.'))
    if match.group(2):
        num *= 1000
    return round(num)

def get_member_count(driver, group_url):
    try:
        driver.get(group_url)
        time.sleep(4)
        elements = driver.find_elements(By.XPATH, '//*[contains(text(), "miembros") or contains(text(), "members")]')
        for el in elements:
            text = el.text.strip()
            if re.search(r'[\d,.]+\s*(mil\s*)?(miembros|members)', text, re.IGNORECASE):
                count = parse_members(text)
                if count:
                    return count
        return None
    except Exception as e:
        print(f'  Error en {group_url}: {e}')
        return None
